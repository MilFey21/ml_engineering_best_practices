    Пример 10: Предсказание на новых данных¶

    import joblib
import pandas as pd
from pathlib import Path
from src.features import create_features

# Загрузка модели
model = joblib.load("models/model.pkl")

# Загрузка новых данных
new_data = pd.read_csv("data/raw/new_customers.csv")

# Создание признаков (используя ту же логику, что и при обучении)
features = create_features(new_data)

# Предсказание
predictions = model.predict(features)
probabilities = model.predict_proba(features)

# Сохранение результатов
results = pd.DataFrame({
    'customer_id': new_data['customerID'],
    'churn_prediction': predictions,
    'churn_probability': probabilities[:, 1]
})
results.to_csv("predictions.csv", index=False)

    Пример 11: Использование контекстного менеджера ClearML¶

    from src.modeling.clearml_utils import clearml_context
from src.modeling.train import train_model

# Обучение с автоматическим логированием
with clearml_context(
    project_name="Churn Prediction",
    task_name="experiment_1",
    tags=["baseline", "random_forest"],
) as exp:
    # Логирование параметров
    exp.log_params({
        "n_estimators": 100,
        "max_depth": 10,
        "random_state": 42,
    })

    # Обучение модели
    metrics = train_model(...)

    # Логирование метрик
    exp.log_metrics(metrics)
